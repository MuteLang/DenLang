' Array & Range example

array content
  first
    : content.1..1
  last
    : content.-1..1

example:array "hello world"

! example.first ' h
! example.last ' d



' Split example

array content
  index : 0
  split char
    ? = content.index char 
      : content.0..index
    + index 1

example:array "hello world"

! example.split " " ' hello
